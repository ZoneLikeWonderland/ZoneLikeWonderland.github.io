### B. Just Eat It!
给定数列a[i]，问是否存在某个区间（不能是1~n）元素之和大于等于所有元素之和。
等价于是否存在两端开始的任意元素之和小于等于0。
```c++
#include <bits/stdc++.h>
#define fx(i, a, b) for (register int i = int(a); i <= int(b); i++)
typedef long long ll;
using namespace std;
const int N = 1e5 + 7;
ll a[N];
int main() {
    int T;
    cin >> T;
    while (T--) {
        int n;
        cin >> n;
        fx(i, 1, n) cin >> a[i];
        ll l = 0;
        ll r = 0;
        int yes = 0;
        fx(i, 1, n - 1) {
            l += a[i];
            if (l <= 0) {
                yes = 1;
                break;
            }
        }
        for (int i = n; i >= 2; i--) {
            r += a[i];
            if (r <= 0) {
                yes = 1;
                break;
            }
        }
        if (!yes)
            cout << "YES\n";
        else
            cout << "NO\n";
    }
}
```
### C. Fadi and LCM
给定LCM，求使两个数LCM是它的数中 最大数最小的数对（可能有多个，输出任意一个）。好像想复杂了，暴力应该就能过了。我找到的规律是对LCM分解质因数，最优的情况一定是两个数不共享任意一个质因数。因此对LCM分解质因数，把相同质因数的幂枚举它在第一个数还是第二个数，找到最大值最小的情况。可以证明枚举复杂度很低。
```c++
#include <bits/stdc++.h>
#define fx(i, a, b) for (register ll i = ll(a); i <= ll(b); i++)
typedef long long ll;
using namespace std;
const int N = 1e5 + 7;
ll a[N];
int cnt = 0;
int main() {
    ll n;
    cin >> n;
    ll g = sqrt(n);
    fx(i, 2, g + 1) {
        if (n % i == 0) {
            ++cnt;
            a[cnt] = 1;
            while (n % i == 0) {
                // cout << i << " ";
                n /= i;
                a[cnt] *= i;
            }
        }
    }
    if (n > 1) {
        ll i = n;
        if (n % i == 0) {
            ++cnt;
            a[cnt] = 1;
            while (n % i == 0 && n > 0) {
                // cout << i << " ";
                n /= i;
                a[cnt] *= i;
            }
        }
    }
    ll ans = 1ll << 60;
    ll l = 1, r = 1;
    if (cnt) fx(i, 0, (1 << (cnt - 1)) - 1) {
            ll lt = 1, rt = 1;
            fx(j, 1, cnt) {
                if ((1 << (j - 1)) & i)
                    lt *= a[j];
                else
                    rt *= a[j];
            }
            if (max(lt, rt) < ans) {
                ans = max(lt, rt);
                l = lt, r = rt;
            }
        }
    cout << l << " " << r << "\n";
}
```
### D. Dr. Evil Underscores
给定一些数，求$\underset{1 \leq i \leq n}{\max} (a_i \oplus X)$。
作业里写的是分治
```c++
#include <bits/stdc++.h>
#define fx(i, a, b) for (register ll i = ll(a); i <= ll(b); i++)
typedef long long ll;
using namespace std;

const int N = 1e5 + 7;

vector<ll> a;
int cnt = 0;

int fun(int k, vector<ll> a) {
    vector<ll> b, c;
    for (auto &x : a) {
        // cout<<x<<" ";
        if ((x >> k) & 1)
            b.push_back(x);
        else
            c.push_back(x);
    }

    if (!k) return (b.empty() || c.empty()) ? 0 : 1;
    if (b.empty()) return fun(k - 1, c);
    if (c.empty()) return fun(k - 1, b);
    // cout<<k<<"+1\n";
    return (1ll << k) + min(fun(k - 1, b), fun(k - 1, c));
}

int main() {
    int n;
    cin >> n;
    fx(i, 1, n) {
        ll t;
        cin >> t;
        a.push_back(t);
    }
    cout << fun(32, a);
}
```
### E. Delete a Segment
给定一些线段（可能退化为点），问删除某一个线段后其余能组成的最大覆盖（union）数目。
学习到一种扫描线写法：
```c++
        fx(i, 1, n) {
            ll a, b;
            cin >> a >> b;
            v.push_back(make_pair(a, -i));
            v.push_back(make_pair(b, i));
        }
```
扫描线扫描的时候记录每一条线段连接两个union的次数，如果次数不为零，原始的union数加上这个次数就是答案。如果没有线段连接两个union的话考虑被孤立的线段数目，如果至少有一个线段没有被孤立那么答案就是原始的union数。如果所有都是孤立线段，那么至少会少一个，就是n-1。（这种扫描线方法排序后貌似对多个点重合的情况处理起来很简单。）
```c++
#include <bits/stdc++.h>
#define fx(i, a, b) for (register ll i = ll(a); i <= ll(b); i++)
typedef long long ll;
using namespace std;
const int N = 1e6 + 7;
vector<pair<ll, ll>> v;
unordered_map<ll, int> u;
int w[N];

int main() {
    int T;
    cin >> T;
    while (T--) {
        int n;
        cin >> n;
        u.clear();
        v.clear();
        fx(i, 1, n) {
            ll a, b;
            cin >> a >> b;
            v.push_back(make_pair(a, -i));
            v.push_back(make_pair(b, i));
        }
        memset(w, 0, n * 4 + 100);
        sort(v.begin(), v.end());
        int pos = v[0].first - 1;
        int layer = 0;
        bool p21 = false, all1 = false;
        int split = 0;
        int xuankong = 0;
        for (auto &x : v) {
            if (x.first != pos) {
                if (x.second < 0 && layer == 1 && p21) { w[(*u.begin()).first]++; }
                if (layer == 0) split++;
            }
            pos = x.first;
            p21 = false;
            if (x.second < 0) {
                u[-x.second] = 1;
                layer++;
                if (layer == 1)
                    all1 = true;
                else
                    all1 = false;
            } else {
                u.erase(x.second);
                layer--;
                if (layer == 1) p21 = true;
                if (layer == 0 && all1) xuankong += 1;
            }
        }
        int m = 0;
        fx(i, 1, n) m = max(m, w[i]);
        if (m)
            cout << split + m << "\n";
        else if (n - xuankong)
            cout << split << "\n";
        else
            cout << n - 1 << "\n";
    }
}
```
